\documentclass{minimal}

\usepackage{tikz}
\usetikzlibrary{positioning}
\usetikzlibrary{backgrounds}
\usetikzlibrary{patterns.meta, arrows.meta, bending}

\begin{document}

\tikzset{arena/.style = 
	{
		very thick, 
		fill=gray,
		fill opacity=0.5
	}
}

\tikzset {marker/.style =
	{
		rectangle,
		densely dashed,
		minimum size=0.5cm,
		inner sep=0cm,
		text opacity=1,
		draw
	}
}

\tikzset{aoe/.style =
	{
		thick,
		orange, 
		pattern={
			Lines[angle=45, distance=0.5cm, line width=0.25cm]
		},
		pattern color=yellow,
		opacity=0.25,
		draw opacity=1
	}
}

\tikzset{aoe fill/.style = {
		fill=red,
		opacity=0.25	
	}
}

\tikzset{cone aoe/.pic = {
        \draw [preaction={aoe fill}] [aoe] (0,0) -- (#1/2:10) arc (#1:-1*#1:5) -- cycle;	
    }
}

\tikzset{line aoe/.pic = {
        \draw [preaction={aoe fill}] [aoe] (0,-0.5*#1) rectangle (10,0.5*#1);	
    }
}

\tikzset{circle aoe/.pic = {
        \draw [preaction={aoe fill}] [aoe] (0,0) circle [radius=#1];
    }
}

\tikzset{pics/donut aoe/.style 2 args = {
        code = {
            \draw [preaction={even odd rule,aoe fill}] [even odd rule,aoe] (0,0) circle [radius=#1] circle [radius=#2];
        }
    }
}

\tikzset{boss/.style = 
	{
		shape=circle,
		very thick,
		purple,
		fill=purple!50,
		fill opacity=0.75,
		minimum size=1.5cm,
		text=black,
		text opacity=1,
		draw
	}
}

\tikzset{tank/.style = 
	{
		boss,
		draw=blue,
		fill=blue!50,
		minimum size=0.5cm
	}
}

\tikzset{dps/.style = 
	{
		tank,
		draw=red,
		fill=red!50
	}
}

\tikzset{destination/.style = 
	{
		tank,
		dashed,
		draw=black,
		fill=black!50
	}
}

\tikzset{heal/.style = 
	{
		tank,
		draw=green,
		fill=green!50	
	}
}

\tikzset{nf/.style = { fill=none } }

\tikzset{move/.style = {
		line width=2pt,
		-Latex,
		fill=none
	}
}

\tikzset{tether/.style = {
		line width=2pt,
		purple,
		dotted,
		Latex-Latex,
		text=black
	}
}

\begin{tikzpicture}[scale=1.2]

	\node[boss] (boss) at (0.5,0) {B};
	\node[tank, above left=1cm and 0.5cm of boss] (t1) {1};
	\node[tank, above right=0cm of boss] (t2) {2};
	\node[heal, below left=0.25cm of boss] (h1) {1};
	\node[heal, below left=0.25cm of h1] (h2) {2};
	\node[dps, below=0cm of boss] (d1) {1};
	\node[dps, below right=0cm of boss] (d2) {2};
	\node[dps, on grid, below left=2.5cm and 1cm of boss] (d3) {3};
	\node[dps] at (3,-1) (d4) {4};
	
	\foreach \letter / \ang / \rad / \c in 
	    {A/90/3.5/red,
	    D/180/3.5/purple,
	    C/270/3.5/blue,
	    B/0/3.5/yellow,
	    1/45/2.5/red,
	    4/135/2.5/purple,
	    3/225/2.5/blue,
	    2/315/2.5/yellow}
		\node[marker, color=\c, fill, fill opacity=0.25] at (\ang:\rad) (marker-\letter) {\textbf\letter};
	
	\begin{scope}[on background layer]
		\clip (0,0) circle [radius=4];
		\draw[help lines] (-5,-5) grid +(10,10);
		\draw[arena] (0,0) circle [radius=4];
		\path (boss.center) -- pic [sloped, allow upside down, at start] {cone aoe=30} (t1) ;
		\path (d3.center) -- pic [sloped, allow upside down, at start] {line aoe=1cm} (h2) ;
		\pic at (d4) {circle aoe=1cm};
		\pic at (marker-C) {donut aoe={1cm}{3cm}};
	\end{scope}
	
	\draw[move] (h2) to [bend left] (-3,0) node (h2x) {};
	\node [right=0.5cm of h2x.center] (h1x){};
	\draw[move] (h1) to [bend left] (h1x.center);
	\draw[tether] (t2) -- (d4) node [midway, below, sloped] {\textit{break tether}}; 
	
\end{tikzpicture}
